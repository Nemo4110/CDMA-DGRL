# conda环境安装

```bash
conda create -p /data/data2/041/envs/LERS2 python==3.9.0
conda install pytorch==1.12.1 torchvision==0.13.1 torchaudio==0.12.1 cudatoolkit=11.3
conda install pyg -c pyg
pip install d2l
```
